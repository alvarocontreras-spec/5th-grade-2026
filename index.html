<script>

    let recorders = {};

    let audioChunks = {};

    let timerIntervals = {};

    let secondsCounter = {};

 

    async function startRecording(id) {

        try {

            const stream = await navigator.mediaDevices.getUserMedia({ audio: true });

            const mediaRecorder = new MediaRecorder(stream);

           

            recorders[id] = mediaRecorder;

            audioChunks[id] = [];

            secondsCounter[id] = 0;

 

            mediaRecorder.ondataavailable = (e) => {

                if (e.data.size > 0) audioChunks[id].push(e.data);

            };

 

            mediaRecorder.onstop = () => {

                const blob = new Blob(audioChunks[id], { type: 'audio/webm' });

                const url = URL.createObjectURL(blob);

                const audio = document.getElementById(`audio-${id}`);

                audio.src = url;

                audio.style.display = 'block';

                document.getElementById(`dl-btn-${id}`).style.display = 'inline-block';

                stream.getTracks().forEach(track => track.stop());

            };

 

            mediaRecorder.start();

            startTimer(id);

            updateUI(id, 'recording');

        } catch (err) {

            alert("Please allow microphone access.");

        }

    }

 

    function togglePause(id) {

        const mr = recorders[id];

        const btn = document.getElementById(`pause-btn-${id}`);

        if (mr.state === "recording") {

            mr.pause();

            clearInterval(timerIntervals[id]);

            updateUI(id, 'paused');

            btn.textContent = "▶ Resume";

        } else {

            mr.resume();

            startTimer(id);

            updateUI(id, 'recording');

            btn.textContent = "⏸ Pause";

        }

    }

 

    function stopRecording(id) {

        recorders[id].stop();

        clearInterval(timerIntervals[id]);

        updateUI(id, 'stopped');

    }

 

    function startTimer(id) {

        timerIntervals[id] = setInterval(() => {

            secondsCounter[id]++;

            const m = Math.floor(secondsCounter[id] / 60).toString().padStart(2, '0');

            const s = (secondsCounter[id] % 60).toString().padStart(2, '0');

            document.getElementById(`timer-${id}`).textContent = `${m}:${s}`;

        }, 1000);

    }

 

    function updateUI(id, state) {

        const status = document.getElementById(`status-${id}`);

        const recBtn = document.getElementById(`rec-btn-${id}`);

        const pauseBtn = document.getElementById(`pause-btn-${id}`);

        const stopBtn = document.getElementById(`stop-btn-${id}`);

 

        if (state === 'recording') {

            status.textContent = "Recording...";

            status.className = "status-indicator status-recording";

            recBtn.disabled = true;

            pauseBtn.disabled = false;

            stopBtn.disabled = false;

        } else if (state === 'paused') {

            status.textContent = "Paused";

            status.className = "status-indicator status-paused";

        } else if (state === 'stopped') {

            status.textContent = "Finished";

            status.className = "status-indicator status-stopped";

            pauseBtn.disabled = true;

            stopBtn.disabled = true;

        }

    }

 

    function downloadAudio(id) {

        const name = document.getElementById('studentName').value || 'Student';

        const blob = new Blob(audioChunks[id], { type: 'audio/webm' });

        const url = URL.createObjectURL(blob);

        const a = document.createElement('a');

        a.href = url;

        a.download = `${name}_Section_${id}.webm`;

        a.click();

    }

 

    function generateReport() {

        const name = document.getElementById('studentName').value;

        const group = document.getElementById('className').value;

        if(!name || !group) return alert("Fill in Name and Class first!");

 

        document.getElementById('assessmentForm').style.display = 'none';

        document.getElementById('reportContainer').classList.add('show');

        document.getElementById('rep-info').textContent = `Student: ${name} | Class: ${group} | Date: ${new Date().toLocaleDateString()}`;

       

        // Simular notas para el reporte visual

        setTimeout(() => {

            document.getElementById('bar1').style.width = "85%";

            document.getElementById('bar2').style.width = "78%";

        }, 500);

    }

</script>
